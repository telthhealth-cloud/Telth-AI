tsx
import { auth } from "firebase-config";
import { RecaptchaVerifier, signInWithPhoneNumber } from "firebase/auth";
import { useState } from "react";

export default function PhoneAuth() {
  const countryCode = "+91";
  const [phoneNumber, setPhoneNumber] = useState(countryCode);
  const [expandForm, setExpandForm] = useState(false);

  const generateRecaptcha = () => {
    if (!window.recaptchaVerifier) {
      window.recaptchaVerifier = new RecaptchaVerifier(
        "recaptcha-container",
        {
          size: "invisible",
          callback: (response: any) => {
            console.log("reCAPTCHA solved:", response);
          },
        },
        auth
      );
    }
  };

  const requestOTP = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();

    const formData = new FormData(e.currentTarget);
    const phone = formData.get("phone")?.toString() || "";

    setPhoneNumber(phone);

    if (phone.length >= 10) {
      setExpandForm(true);
      generateRecaptcha();

      const appVerifier = window.recaptchaVerifier;

      signInWithPhoneNumber(auth, phone, appVerifier)
        .then((confirmationResult) => {
          window.confirmationResult = confirmationResult;
          console.log(" OTP sent:", confirmationResult);
        })
        .catch((error) => {
          console.error(" Error sending OTP:", error);
        });
    } else {
      console.error(" Invalid phone number:", phone);
    }
  };

  const getOTPField = () => (
    <div className="form-group">
      <label>OTP</label>
      <input
        type="text"
        className="w-2xl h-8 rounded-md border p-2"
        name="otp"
      />
    </div>
  );

  return (
    <main className="min-w-screen min-h-screen flex items-center justify-center">
      <form onSubmit={requestOTP} className="flex flex-col gap-3">
        <h1 className="text-center text-3xl font-semibold">Phone Auth</h1>

        <div className="form-group">
          <label>Phone</label>
          <input
            type="tel"
            id="phoneNumberInput"
            defaultValue={countryCode}
            className="w-2xl h-8 rounded-md border p-2"
            name="phone"
          />
        </div>

        {/*  Proper conditional render */}
        {expandForm && getOTPField()}

        <div id="recaptcha-container"></div>

        <button type="submit" className="py-2 rounded-md bg-indigo-500 text-white">
          Submit
        </button>
      </form>
    </main>
  );
}

The error:

Firebase: Error (auth/billing-not-enabled)

means your Firebase project has phone authentication enabled, but Cloud Functions billing (Blaze plan) is not enabled which is required for Firebase Phone Auth to work in production or certain regions.

Why It Happens

Firebase Phone Authentication uses Google Clouds SMS infrastructure (via Identity Toolkit), and that feature requires a Blaze (pay-as-you-go) plan.

If your Firebase project is on the free Spark plan, youll get:

auth/billing-not-enabled

even if youve enabled Phone Auth in the Authentication section.

How to Fix It (Step-by-Step)

1. Go to the Firebase Console

https://console.firebase.google.com

2. Open Your Project

3. Go to Project Settings Usage and billing

4. Click Upgrade to Blaze plan

  • The Blaze plan is pay as you go but youll only be billed if you exceed the free limits (Firebase Auth SMS verification gives you a small free quota per month).

  • Youll need to add a billing account (credit/debit card).

5. Confirm Phone Authentication is enabled

  • In Authentication Sign-in Method Phone, ensure its enabled.

6. (Optional but important) Set up a reCAPTCHA key if using web.

Firebase automatically handles invisible reCAPTCHA, but make sure:

  • Youre serving your site over HTTPS

  • Your domain is added under Authentication Settings Authorized domains

Local Development Tip

If youre just testing locally and dont want to enable billing yet:

  1. Use Firebase Auth emulator (no billing required).
    Run in your terminal:

  2. Add emulator connection in your app:

    ts

connectAuthEmulator(auth, "http://localhost:9099");

  3. You can then test SMS verification with fake numbers like:

    diff
    +15555555555

    (Firebase emulator has built-in test numbers.)

Lets break this down carefully

Common Causes (and Fixes)

1. Incorrect RecaptchaVerifier initialization

In your code, you had:

ts
window.recaptchaVerifier = new RecaptchaVerifier("recaptcha-container", { size: "invisible" }, auth);

This is correct but you must ensure:

  • The auth object comes from the same Firebase app where you initialized Firebase.

  • The container ID ("recaptcha-container") exists in the DOM before you call new RecaptchaVerifier().

If you call it before the DOM renders, Firebase cant attach the widget and youll get auth/invalid-app-credential.

Fix:
Ensure the reCAPTCHA container exists when you call it:

ts
const generateRecaptcha = () => {
  if (!window.recaptchaVerifier) {
    window.recaptchaVerifier = new RecaptchaVerifier(
      "recaptcha-container",
      { size: "invisible" },
      auth
    );
  }
};

Make sure this is only called after the form renders (for example, right before you call signInWithPhoneNumber()).

2. Wrong Firebase configuration or auth instance

Make sure your Firebase setup looks like this:

ts
// firebase-config.ts
import { initializeApp } from "firebase/app";
import { getAuth } from "firebase/auth";

const firebaseConfig = {
  apiKey: "YOUR_API_KEY",
  authDomain: "your-project.firebaseapp.com",
  projectId: "your-project-id",
  storageBucket: "your-project.appspot.com",
  messagingSenderId: "123456789",
  appId: "YOUR_APP_ID"
};

const app = initializeApp(firebaseConfig);
export const auth = getAuth(app);

Then import it correctly:

ts
import { auth } from "./firebase-config";

If your auth instance comes from a different Firebase app, the credential signature wont match and Firebase will reject it auth/invalid-app-credential.

3. Missing or misconfigured Authorized Domains

Firebase checks that the request comes from a domain listed in your project settings.

Go to:
Firebase Console Authentication Settings Authorized domains

Make sure your current domain is listed:

  • localhost (for local testing)

  • Your production domain (e.g. myapp.web.app, myapp.firebaseapp.com, etc.)

If not add it manually.

4. Calling signInWithPhoneNumber twice without regenerating the reCAPTCHA

Each call to signInWithPhoneNumber uses a single-use reCAPTCHA token.
If you reuse the same verifier instance without regenerating it after an error, youll get this same auth/invalid-app-credential.

Fix: regenerate the reCAPTCHA before each call.

Example:

ts
const appVerifier = window.recaptchaVerifier;
if (!appVerifier) generateRecaptcha(); // ensure its fresh

5. Using Emulator vs. Production mismatch

If youre using the Auth emulator, dont call signInWithPhoneNumber with reCAPTCHA it doesnt work there.

Instead, connect to the emulator and use test numbers:

ts
import { connectAuthEmulator } from "firebase/auth";
connectAuthEmulator(auth, "http://localhost:9099");
